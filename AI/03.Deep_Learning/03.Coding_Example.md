class TextPreprocessor {
    constructor() {
        this.vocabulary = {};
        this.wordIndex = 1; // Start from 1, reserve 0 for padding/unknown
    }

    // Build vocabulary from training texts
    buildVocabulary(texts) {
        console.log("📚 Building vocabulary from training data...");
        
        texts.forEach(text => {
            const words = this.tokenize(text);
            words.forEach(word => {
                if (!this.vocabulary[word]) {
                    this.vocabulary[word] = this.wordIndex++;
                }
            });
        });
        
        console.log(`   ✓ Vocabulary size: ${Object.keys(this.vocabulary).length} words`);
        console.log(`   ✓ Sample words:`, Object.keys(this.vocabulary).slice(0, 10));
        console.log();
    }

    // Tokenize text into words
    tokenize(text) {
        return text
            .toLowerCase()
            .replace(/[^\w\s]/g, '') // Remove punctuation
            .split(/\s+/)
            .filter(word => word.length > 0);
    }

    // Convert text to sequence of word indices
    textToSequence(text) {
        const words = this.tokenize(text);
        return words.map(word => this.vocabulary[word] || 0); // 0 for unknown words
    }

    // Convert text to bag of words vector
    textToBagOfWords(text, vocabSize) {
        const vector = new Array(vocabSize).fill(0);
        const words = this.tokenize(text);
        
        words.forEach(word => {
            const index = this.vocabulary[word];
            if (index && index < vocabSize) {
                vector[index] = 1; // Binary: word present or not
            }
        });
        
        return vector;
    }
}

// ============================================================================
// STEP 2: NEURAL NETWORK - THE BRAIN
// ============================================================================

class NeuralNetwork {
    constructor(inputSize, hiddenSize, outputSize) {
        this.inputSize = inputSize;
        this.hiddenSize = hiddenSize;
        this.outputSize = outputSize;
        
        // Initialize weights with random values
        this.weightsInputHidden = this.randomMatrix(inputSize, hiddenSize);
        this.weightsHiddenOutput = this.randomMatrix(hiddenSize, outputSize);
        
        // Initialize biases
        this.biasHidden = new Array(hiddenSize).fill(0).map(() => Math.random() * 0.01);
        this.biasOutput = new Array(outputSize).fill(0).map(() => Math.random() * 0.01);
        
        this.learningRate = 0.1;
    }

    // Create random matrix for initial weights
    randomMatrix(rows, cols) {
        const matrix = [];
        for (let i = 0; i < rows; i++) {
            matrix[i] = [];
            for (let j = 0; j < cols; j++) {
                // Xavier initialization
                matrix[i][j] = (Math.random() - 0.5) * 2 * Math.sqrt(6 / (rows + cols));
            }
        }
        return matrix;
    }

    // Activation function: Sigmoid
    sigmoid(x) {
        return 1 / (1 + Math.exp(-x));
    }

    // Derivative of sigmoid
    sigmoidDerivative(x) {
        return x * (1 - x);
    }

    // Activation function: ReLU
    relu(x) {
        return Math.max(0, x);
    }

    // Derivative of ReLU
    reluDerivative(x) {
        return x > 0 ? 1 : 0;
    }

    // Forward propagation
    forward(input) {
        // Input to Hidden layer
        this.input = input;
        this.hidden = [];
        
        for (let j = 0; j < this.hiddenSize; j++) {
            let sum = this.biasHidden[j];
            for (let i = 0; i < this.inputSize; i++) {
                sum += input[i] * this.weightsInputHidden[i][j];
            }
            this.hidden[j] = this.sigmoid(sum); // Using sigmoid for hidden layer
        }

        // Hidden to Output layer
        this.output = [];
        for (let k = 0; k < this.outputSize; k++) {
            let sum = this.biasOutput[k];
            for (let j = 0; j < this.hiddenSize; j++) {
                sum += this.hidden[j] * this.weightsHiddenOutput[j][k];
            }
            this.output[k] = this.sigmoid(sum); // Using sigmoid for output
        }

        return this.output;
    }

    // Backward propagation (learning from mistakes)
    backward(input, target) {
        // Calculate output layer error
        const outputErrors = [];
        for (let k = 0; k < this.outputSize; k++) {
            outputErrors[k] = target[k] - this.output[k];
        }

        // Calculate output layer gradients
        const outputGradients = [];
        for (let k = 0; k < this.outputSize; k++) {
            outputGradients[k] = outputErrors[k] * this.sigmoidDerivative(this.output[k]);
        }

        // Calculate hidden layer error
        const hiddenErrors = [];
        for (let j = 0; j < this.hiddenSize; j++) {
            let error = 0;
            for (let k = 0; k < this.outputSize; k++) {
                error += outputGradients[k] * this.weightsHiddenOutput[j][k];
            }
            hiddenErrors[j] = error;
        }

        // Calculate hidden layer gradients
        const hiddenGradients = [];
        for (let j = 0; j < this.hiddenSize; j++) {
            hiddenGradients[j] = hiddenErrors[j] * this.sigmoidDerivative(this.hidden[j]);
        }

        // Update weights between hidden and output
        for (let j = 0; j < this.hiddenSize; j++) {
            for (let k = 0; k < this.outputSize; k++) {
                this.weightsHiddenOutput[j][k] += this.learningRate * outputGradients[k] * this.hidden[j];
            }
        }

        // Update output biases
        for (let k = 0; k < this.outputSize; k++) {
            this.biasOutput[k] += this.learningRate * outputGradients[k];
        }

        // Update weights between input and hidden
        for (let i = 0; i < this.inputSize; i++) {
            for (let j = 0; j < this.hiddenSize; j++) {
                this.weightsInputHidden[i][j] += this.learningRate * hiddenGradients[j] * input[i];
            }
        }

        // Update hidden biases
        for (let j = 0; j < this.hiddenSize; j++) {
            this.biasHidden[j] += this.learningRate * hiddenGradients[j];
        }

        // Calculate loss (Mean Squared Error)
        let loss = 0;
        for (let k = 0; k < this.outputSize; k++) {
            loss += Math.pow(outputErrors[k], 2);
        }
        return loss / this.outputSize;
    }

    // Train the network
    train(inputs, targets, epochs) {
        console.log("🧠 Training Neural Network...");
        console.log(`   Network architecture: ${this.inputSize} → ${this.hiddenSize} → ${this.outputSize}`);
        console.log(`   Training for ${epochs} epochs`);
        console.log();

        for (let epoch = 0; epoch < epochs; epoch++) {
            let totalLoss = 0;

            for (let i = 0; i < inputs.length; i++) {
                this.forward(inputs[i]);
                const loss = this.backward(inputs[i], targets[i]);
                totalLoss += loss;
            }

            const avgLoss = totalLoss / inputs.length;

            // Print progress every 100 epochs
            if (epoch % 100 === 0 || epoch === epochs - 1) {
                console.log(`   Epoch ${epoch + 1}/${epochs} - Loss: ${avgLoss.toFixed(6)}`);
            }
        }

        console.log("   ✓ Training completed!");
        console.log();
    }

    // Make prediction
    predict(input) {
        const output = this.forward(input);
        return output;
    }
}

    console.log(`   ✓ Prediction: ${sentiment} (Confidence: ${confidence}%)`);
    console.log();
    
    return { sentiment, confidence, rawScore: prediction[0] };
}
